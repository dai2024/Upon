    SUBI.B #$30, D0      ; Subtract the ASCII code for '0' from D0 to convert it to a number
    ADD.D D0, D0         ; Multiply the number by 2
    SUBQ #1, D1          ; Decrement D1 to count the number of iterations
    BNE LOOP             ; Branch to LOOP if D1 is not zero
