ORG 100H ; Set the program start address to 0x100

START:
    LXI H, 4300H ; Set the address of i to 4300H
    MVI A, 0     ; Initialize the accumulator to zero
LOOP:
    LDAX H       ; Load the value at address H into the accumulator
    CPI '0'      ; Compare A to the character '0'
    JC END       ; Jump to END if A is less than '0'
    CPI '9'      ; Compare A to the character '9'
    JA END       ; Jump to END if A is greater than '9'
    SUB '0'      ; Subtract the character '0' from A to convert it to a number
    ADD A        ; Add A to itself to multiply it by 2
    DCR C        ; Decrement C to count the number of iterations
    JNZ LOOP     ; Jump to LOOP if C is not zero
    HLT          ; Halt the program

END:
    HLT          ; Halt the program

; Data section
    DB '4', '3'  ; Define the string "43"
    DC 2         ; Define the length of the string as 2
    DS 1         ; Define a byte to store the result
