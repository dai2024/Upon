u16 = Uint16(0x1234)
packed_bytes = u16.pack()
print(packed_bytes)  # b'\x12\x34'
unpacked_u16 = Uint16.unpack(packed_bytes)
print(unpacked_u16.value)  # 4660

u8 = Uint8(0xff)
packed_bytes = u8.pack()
print(packed_bytes)  # b'\xff'
unpacked_u8 = Uint8.unpack(packed_bytes)
print(unpacked_u8.value)  # 255

i16 = Int16(-1234)
packed_bytes = i16.pack()
print(packed_bytes)  # b'\xf7\x2e'
unpacked_i16 = Int16.unpack(packed_bytes)
print(unpacked_i16.value)  # -6162

i8 = Int8(-128)
packed_bytes = i8.pack()
print(packed_bytes)  # b'\x80'
unpacked_i8 = Int8.unpack(packed_bytes)
print(unpacked_i8.value)  # -128
