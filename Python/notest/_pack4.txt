import struct

class ExampleStruct:
    def __init__(self, c1: int, i1: int, c2: int):
        self.c1 = c1
        self.i1 = i1
        self.c2 = c2

    def pack(self):
        return struct.pack("<BIQb", self.c1, self.i1, self.c2)

#
if __name__ == "__main__":
    s = ExampleStruct(0x01, 0x01234567, 0x02)
    packed = s.pack()
    print(packed.hex())
