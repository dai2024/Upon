#
def calc_padding(size):
    padding = 2 - (size % 2)
    return padding if padding < 2 else 0

#
def pack_data(data):
    #
    padding = calc_padding(len(data))

    #
    packed_data = data + b'\0' * padding

    #
    return struct.pack('!' + str(len(packed_data)) + 's', packed_data)

#
