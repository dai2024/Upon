class Uint8:
    def __init__(self, w0):
        self.w0 = w0

def multiply_uint8(x, y):
    x = Uint8(x & 0xff)
    y = Uint8(y & 0xff)

    z = Uint8(0)

    for i in range(8):
        for j in range(8):
            if i + j == 8:
                pass
            elif i + j < 8:
                z.w0 += (x.w0 >> i) * (y.w0 >> j)

    return z.w0
